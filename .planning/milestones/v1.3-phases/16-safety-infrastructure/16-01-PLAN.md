1. In `internal/executor/executor.go`, add an `isAMSIBlocked` function (not exported — internal to executor):

```go
import "os/exec"

func isAMSIBlocked(errOut string, err error) bool {
    amsiPatterns := []string{
        "ScriptContainedMaliciousContent",
        "This script contains malicious content",
        "has been blocked by your antivirus software",
    }
    for _, p := range amsiPatterns {
        if strings.Contains(errOut, p) {
            return true
        }
    }
    if exitErr, ok := err.(*exec.ExitError); ok {
        if exitErr.ExitCode() == -196608 {
            return true
        }
    }
    return false
}
```

2. In `runInternal`, after the `runCommand`/`runCommandAs` calls (line ~131-135 area, after `out, errOut, err` are assigned but BEFORE `result.Success = (err == nil)`), add the AMSI check — only for PowerShell executors:

```go
// AMSI detection — PowerShell only (per D-02)
if strings.ToLower(t.Executor.Type) == "powershell" || strings.ToLower(t.Executor.Type) == "psh" {
    if isAMSIBlocked(errOut, err) {
        result.Output = out
        result.ErrorOutput = errOut
        result.Success = false
        result.EndTime = time.Now().Format(time.RFC3339)
        result.RunAsUser = userLabel
        result.VerificationStatus = playbooks.VerifAMSIBlocked
        simlog.Info(fmt.Sprintf("[AMSI] %s — blocked by antimalware scan", t.ID))
        simlog.TechOutput(t.ID, out, errOut)
        simlog.TechEnd(t.ID, false, time.Since(start))
        return result
    }
}
```

CRITICAL: This must return early BEFORE the normal `result.Success = (err == nil)` line. The early return prevents the verifier from running (verifier is in `runTechnique`, not `runInternal`), avoiding a misleading "Fail" for event verification on a technique that never executed.
