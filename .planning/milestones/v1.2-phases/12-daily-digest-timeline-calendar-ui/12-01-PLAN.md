3. **renderDayCalendar(days)** — builds the horizontal day strip.

Per D-08: group days by consecutive phase runs. Per D-09: color-code via CSS classes. Per D-10: day number in cell, tooltip with technique count + pass/fail. Per CAL-04: phase labels above groups. Per D-11: click handler on active/complete cells calls `focusDayInDigest(dayNum)`.

Phase grouping logic (per RESEARCH.md Pitfall 6):
```javascript
function renderDayCalendar(days) {
  // Build phase groups: [{phase, days: [...]}, ...]
  const groups = [];
  days.forEach(d => {
    if (!groups.length || groups[groups.length-1].phase !== d.phase) {
      groups.push({ phase: d.phase, days: [] });
    }
    groups[groups.length-1].days.push(d);
  });

  const html = groups.map(g => {
    const label = PHASE_LABEL[g.phase] || g.phase;
    const cells = g.days.map(d => {
      // Status class: gap days use 'gap' class regardless of status
      const cls = d.phase === 'gap' ? 'gap' : (d.status || 'pending');
      // Tooltip per UI-SPEC: "Day N: X techniques | Y pass / Z fail" or "Day N: pending"
      const tip = d.status === 'pending'
        ? 'Day ' + d.day + ': pending'
        : 'Day ' + d.day + ': ' + d.technique_count + ' techniques | ' + d.pass_count + ' pass / ' + d.fail_count + ' fail';
      // Click handler: only active/complete (per D-11)
      const click = (cls === 'active' || cls === 'complete')
        ? ' onclick="focusDayInDigest(' + d.day + ')"'
        : '';
      return '<div class="day-cell ' + cls + '" title="' + tip + '"' + click + '>' + d.day + '</div>';
    }).join('');
    return '<div class="day-group"><div class="day-group-label">' + label + '</div><div class="day-cells">' + cells + '</div></div>';
  }).join('');

  document.getElementById('dayCalendarContent').innerHTML = html;
}
```

4. **renderDayDigest(days)** — builds the accordion list.

Per D-05: only show active + complete days (filter out pending). Per D-06: newest-first sort (reverse by day number). Per D-03: collapsed view shows day number, phase badge, status label, pass/fail, time window. Expanded view shows technique count, start time, last heartbeat. Per D-04: active day auto-expands. Per RESEARCH.md Pitfall 1: preserve expanded state across re-renders.

```javascript
function renderDayDigest(days) {
  // Per D-05: only active + complete
  const visible = days.filter(d => d.status === 'active' || d.status === 'complete');
  // Per D-06: newest-first
  visible.sort((a, b) => b.day - a.day);

  // Per RESEARCH.md Pitfall 1: preserve open state before rebuild
  const wasOpen = new Set();
  document.querySelectorAll('.digest-body.open').forEach(el => {
    const m = el.id.match(/digest-body-(\d+)/);
    if (m) wasOpen.add(parseInt(m[1]));
  });

  const html = visible.map(d => {
    const statusLabel = DAY_STATUS_LABEL[d.status] || d.status;
    const phaseClass = 'phase-poc_' + d.phase;
    const activeClass = d.status === 'active' ? ' active-day' : '';
    const timeWindow = fmtWindow(d.start_time, d.end_time);

    return '<div class="digest-row' + activeClass + '" id="digest-day-' + d.day + '">'
      + '<div class="digest-header" onclick="toggleDigestRow(' + d.day + ')">'
      + '<span class="digest-chevron" id="digest-chevron-' + d.day + '">\u25b6</span>'
      + '<span class="phase-badge ' + phaseClass + '" style="font-size:11px;padding:2px 8px;">Day ' + d.day + '</span>'
      + '<span style="font-size:12px;color:' + (d.status === 'active' ? 'var(--accent)' : 'var(--green)') + ';">' + statusLabel + '</span>'
      + '<span style="font-size:12px;color:var(--green);">\u2713 ' + d.pass_count + '</span>'
      + '<span style="font-size:12px;color:var(--red);">\u2717 ' + d.fail_count + '</span>'
      + '<span style="font-size:12px;color:var(--muted);margin-left:auto;">' + timeWindow + '</span>'
      + '</div>'
      + '<div class="digest-body" id="digest-body-' + d.day + '">'
      + '<div style="font-size:12px;color:var(--text);margin-bottom:6px;">Techniques: ' + d.technique_count + '</div>'
      + '<div style="font-size:12px;color:var(--muted);margin-bottom:4px;">Started: ' + fmtTime(d.start_time) + '</div>'
      + '<div style="font-size:12px;color:var(--muted);">Last heartbeat: ' + fmtTime(d.last_heartbeat) + '</div>'
      + '</div>'
      + '</div>';
  }).join('');

  document.getElementById('dayDigestContent').innerHTML = html;

  // Per D-04: auto-expand active day
  visible.forEach(d => {
    if (d.status === 'active') openDigestRow(d.day);
  });
  // Re-apply previously expanded rows (Pitfall 1)
  wasOpen.forEach(dayNum => openDigestRow(dayNum));
}
```

```javascript
function toggleDigestRow(dayNum) {
  const body = document.getElementById('digest-body-' + dayNum);
  const chevron = document.getElementById('digest-chevron-' + dayNum);
  if (body) body.classList.toggle('open');

function openDigestRow(dayNum) {
  const body = document.getElementById('digest-body-' + dayNum);
  const chevron = document.getElementById('digest-chevron-' + dayNum);
  if (body) body.classList.add('open');

// Per D-11: calendar click scrolls to and expands digest row
function focusDayInDigest(dayNum) {
  const el = document.getElementById('digest-day-' + dayNum);
  if (!el) return; // Guard per RESEARCH.md Pitfall 4
  openDigestRow(dayNum);
  el.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
}
```

```javascript
function updateDayPanels(days) {
  const calPanel = document.getElementById('dayCalendarPanel');
  const digPanel = document.getElementById('dayDigestPanel');
  if (days.length > 0) {
    hasDayData = true;
    calPanel.style.display = 'block';
    digPanel.style.display = 'block';
    renderDayCalendar(days);
    renderDayDigest(days);
  } else {
    hasDayData = false;
    calPanel.style.display = 'none';
    digPanel.style.display = 'none';
  }
}
```
